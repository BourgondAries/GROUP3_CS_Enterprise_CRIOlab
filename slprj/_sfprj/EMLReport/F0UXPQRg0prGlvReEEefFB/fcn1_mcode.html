1function [tau, s_dot] = fcn(eta, nu, eta_d, eta_ds, eta_dss, Us, Uss, mu, Kp)
 2
 3x = eta(1);
 4y = eta(2);
 5psi = eta(3);
 6
 7u = nu(1);
 8v = nu(2);
 9r = nu(3);
10
11Xu = -0.6555;  Yv = -1.33;    Nv = 0.0; 
12Xuu = 0.3545;  Yvv = -2.776;  Nvv = -0.2088; 
13Xuuu = -3.787; Yvvv = -64.91; Nvvv = 0.0;
14Xv = 0.0;      Yr = -7.25;    Nr = -1.9; 
15Xvv = -2.443;  Yrr = -3.45;   Nrr = 0.75;
16Xvvv = 0.0;    Yrrr = 0.0;    Nrrr = 0.0;
17                Yrv = 0.805;   Nrv = 0.130;
18                Yvr = -0.845;  Nvr = 0.080;
19                
20m = 14.79; 
21Iz = 1.76; 
22xg = 0.0375; 
23yg = 0.0; 
24
25Xudot = -2; 
26Yvdot = -10; 
27Yrdot = -0; 
28Nrdot = -1; 
29
30                
31d11 = -Xu - Xuu * abs(u) - Xuuu*u^2;
32d22 = -Yv - Yvv * abs(v) - Yvvv*v^2 - Yrv*abs(r);
33d23 = -Yr - Yvr*abs(v) - Yrr * abs(r) - Yrrr*r^2;
34d32 = -Nv - Nvv*abs(v) - Nvvv*v^2 - Nrv*abs(r);
35d33 = -Nr - Nvr*abs(v) - Nrr * abs(r) - Nrrr*r^2;

39a_31 = 0;           a_32 = m*xg - Yrdot;     a_33 = Iz - Nrdot;

41M = [a_11 a_12, a_13; a_21 a_22 a_23; a_31 a_32 a_33];
42
43D = [d11, 0, 0;
44    0, d22, d23;
45    0, d32, d33];
46

48    sin(psi), cos(psi), 0;

50
51z1 = R'*(eta-eta_d);

53

55

58z1dot = S'*z1 + nu - s_dot*R'*eta_ds;
59alpha_dot = -Kp*z1dot + S'*R'*eta_ds*Us + R'*eta_dss*s_dot*Us + R'*eta_ds*Uss;
60
61tau = D*nu + M*alpha_dot - z1;
